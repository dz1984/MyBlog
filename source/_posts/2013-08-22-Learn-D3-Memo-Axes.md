### 增加X軸
SVG物件插入一個g元素，用D3的call()去建立X軸。
{% codeblock lang:js %}
var xAxis = d3.svg.axis()
	.scale(xScale)
	.orient("bottom")
	.ticks(5);
	
svg.append('g')
	.attr('class','axis')
	.attr('transform','translate(0,'+(h-padding)+')')
	.call(xAxis);
{% endcodeblock %}
